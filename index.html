⚙️ Google Sheet সংযোগ

আপনার Google Sheet-এর Apps Script Web App URL এবং Contact Sheet URL দিন। এই তথ্য ব্রাউজারে সেভ থাকবে।

📋 Google Sheet কলাম গঠন:
Name | Phone | Password | Package | Bill | Status | Payment_Link

Apps Script কোড (গ্রাহক):
function doGet(e){
 var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Customers");
 var data = sheet.getDataRange().getValues();
 var headers = data[0];
 var ph = e.parameter.phone;
 var pw = e.parameter.password;
 for(var i=1;i<data.length;i++){
  if(data[i][1]==ph && data[i][2]==pw){
   var obj={};
   headers.forEach((h,j)=>obj[h]=data[i][j]);
   return ContentService.createTextOutput(JSON.stringify({success:true,data:obj})).setMimeType(ContentService.MimeType.JSON);
  }
 }
 return ContentService.createTextOutput(JSON.stringify({success:false})).setMimeType(ContentService.MimeType.JSON);
}
